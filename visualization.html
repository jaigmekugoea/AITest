Python 实现

def bubble_sort(arr):
    # 获取数组长度
    n = len(arr)
    
    # 外层循环控制排序轮数
    for i in range(n):
        # 标记本轮是否发生交换
        swapped = False
        
        # 内层循环进行相邻元素比较
        for j in range(0, n - i - 1):
            # 如果前面的元素大于后面的元素，交换它们
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        
        # 如果本轮没有发生交换，说明已经有序
        if not swapped:
            break
    
    return arr

.NET Core C# 实现

using System;
using System.Linq;

namespace AlgorithmVisualization
{
    class BubbleSort
    {
        public static void Sort(int[] arr)
        {
            int n = arr.Length;
            
            // 外层循环控制排序轮数
            for (int i = 0; i < n; i++)
            {
                bool swapped = false;
                
                // 内层循环进行相邻元素比较
                for (int j = 0; j < n - i - 1; j++)
                {
                    // 如果前面的元素大于后面的元素，交换它们
                    if (arr[j] > arr[j + 1])
                    {
                        int temp = arr[j];
                        arr[j] = arr[j + 1];
                        arr[j + 1] = temp;
                        swapped = true;
                    }
                }
                
                // 如果本轮没有发生交换，说明已经有序
                if (!swapped)
                    break;
            }
        }
        
        static void Main(string[] args)
        {
            int[] data = { 64, 34, 25, 12, 22, 11, 90 };
            Console.WriteLine($"原始数组: {string.Join(", ", data)}");
            Sort(data);
            Console.WriteLine($"排序后: {string.Join(", ", data)}");
        }
    }
}
